API Example 
// Ladybug Panel ZW API Example
// Created by Mr Ntando Ofc

const apiKey = 'your-api-key';
const apiUrl = 'https://ladybugpanelzw.com/api/v2';

// Place an order
const placeOrder = async (serviceId, link, quantity) => {
  const response = await fetch(`${apiUrl}/add`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      key: apiKey,
      action: 'add',
      service: serviceId,
      link: link,
      quantity: quantity
    })
  });
  
  const result = await response.json();
  return result;
};

// Get order status
const getOrderStatus = async (orderId) => {
  const response = await fetch(`${apiUrl}/status`, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      key: apiKey,
      action: 'status',
      order: orderId
    })
  });
  
  return await response.json();
};
